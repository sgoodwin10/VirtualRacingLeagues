interface Driver {
  id: number;
  name: string;
  number: number;
  team: Team;
  points: number;
}

const getDriverStandings = async (seasonId: number): Promise<Driver[]> => {
  const response = await fetch(`/api/seasons/${seasonId}/standings`);
  return response.json();
};
//PHP_EXAMPLE
PHP 
class DriverController extends Controller
{
    public function index(): JsonResponse
    {
        $drivers = Driver::with('team')
            ->orderBy('points', 'desc')
            ->get();

        return response()->json($drivers);
    }
}
